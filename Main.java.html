public class Main {

    static int countEvenNum(int[] arr, int size) { //syntax for declaring a method in java
        int count = 0;
        for(int i = 0; i < size; i++) {
            if(arr[i] % 2 == 0) {
                count++;
            }
        }

        return count;

    }

    public static void main(String[] args) {

        int n = 5;
        int[] yar = {0, 1, 2, 3, 4};

        int m = 100;
        int[] bar = new int[m]; //this is how to declare arrays with size in Java!

        for(int i = 0; i < m; i++) {
            bar[i] = i;
        }

        System.out.println("The total number of even numbers in yar are: " + countEvenNum(yar, n)); //Put the variable with a plus sign to print
        System.out.println("The total number of even numbers in bar are: " + countEvenNum(bar, m)); //When passing arrays as a parameter, we just put their nam
    }
}
